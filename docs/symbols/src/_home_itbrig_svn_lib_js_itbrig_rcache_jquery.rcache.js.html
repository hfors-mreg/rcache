 33 (function($){
 34 
 35     /**
 36      * @desc Active and observable ajax cache for jQuery REST applications. Uses
 37      * jQuery.ajax when interacting with the server.
 38      * <p>A note on keys: Cache keys are case sensitive urls. Take care to use
 39      * consistent urls, even query parameters and hashes count.</p>
 40      * @author Hannes Forsgård
 41      * @class
 42      */
 43     $.rcache = new function(){
 44 
 45 
 46         /* Settings */
 47 
 48 
 49         /**
 50          * @desc rcache settings. See setup() for altering settings.
 51          * @var {dict} settings
 52          */
 53         this.settings = {
 54             'update-on-read': true
 55         };
 56 
 57 
 58         /**
 59          * @desc Set rcache settings. The following settings are available:
 60          * <dl>
 61          *     <dt>update-on-read</dt>
 62          *     <dd>Boolean. Send conditional get requests in the
 63          *     background on cache reads? Deafaults to true.</dd>
 64          * </dl>
 65          * @param {dict} settings
 66          * @returns {$.rcache} Return rcache instance for chaining purposes
 67          */
 68         this.setup = function(settings){
 69             $.extend(this.settings, settings);
 70             return this;
 71         }
 72 
 73         
 74         /* Basic functionality */        
 75         
 76 
 77         /**
 78          * @desc The internal cache store
 79          * @var {dict} cache
 80          */
 81         this.cache = {};
 82 
 83 
 84         /**
 85          * @desc Get number of items in cache
 86          * @returns {int}
 87          */
 88         this.count = function(){
 89             var count = 0;
 90             $.each(this.cache, function() { 
 91                 count++;
 92             });
 93             return count;
 94         }
 95 
 96 
 97         /**
 98          * @desc Returns TRUE if cache includes item, FALSE otherwise
 99          * @param {string} key
100          * @returns {bool}
101          */
102         this.has = function(key){
103             return key in this.cache;
104         }
105 
106 
107         /**
108          * @desc Get item for cache key. If item does not exist
109          * an empty item is created.
110          * @param {string} key
111          * @returns {Item}
112          */
113         this.item = function(key){
114             if ( !this.has(key) ) {
115                 this.cache[key] = new Item(key);
116             }
117             return this.cache[key];
118         }
119 
120 
121         /**
122          * @desc Clear cache
123          * @returns {$.rcache} Return rcache instance for chaining purposes
124          */
125        this.clear = function(){
126             this.cache = {};
127             return this;
128         }
129 
130 
131         /* AJAX */
132 
133 
134         /**
135          * @desc Update all items in cache using conditional GET requests
136          * @param {dict} options Additional jQuery.ajax options
137          * @returns {$.rcache} Return rcache instance for chaining purposes
138          */
139         this.updateAll = function(options){
140             $.each(this.cache, function(key, item) { 
141                 item.update(options);
142             });
143             return this;
144         }
145 
146 
147         /**
148          * @desc Settings used when creating ajax objects
149          * @var {dict} ajaxOpts
150          */
151         this.ajaxOpts = {
152             //Default settings goes here ...
153         }; 
154 
155 
156         /**
157          * @desc Set default ajax optins
158          * @param {dict} options
159          * @returns {$.rcache} Return rcache instance for chaining purposes
160          */
161         this.ajaxSetup = function(options){
162             $.extend(this.ajaxOpts, options);
163             return this;
164         }
165 
166 
167         /**
168          * @desc Internal method to create and fire a jqXHR.
169          * @param {dict} options
170          * @returns {jqXHR}
171          */
172         this.getJqXHR = function(options){
173             var settings = {};
174             $.extend(settings, this.ajaxOpts, options);
175             return $.ajax(settings);
176         }
177 
178     }

182     /**
183      * @desc Each item in cache is represented by an Item object
184      * @param {string} url Url of created item
185      * @class
186      * @name Item
187      */
188     function Item(url){
189 
190         /**
191          * @desc Item url
192          * @name Item.url
193          * @type string
194          */
195         this.url = url;
196 
197 
198         /**
199          * @desc Resource item data
200          * @name Item.data
201          * @type mixed
202          */
203         this.data = false;
204 
205  
206         /**
207          * @desc Request object that fetched resource
208          * @name Item.jqXHR
209          * @type jqXHR
210          */
211         this.jqXHR = false;
212 
213 
214         /**
215          * @desc Callback functions for write events
216          * @name Item.writeCallbacks
217          * @type array
218          */
219         this.writeCallbacks = [];
220         
221 
222         /**
223          * @desc Callback functions for remove events
224          * @name Item.removeCallbacks
225          * @type array
226          */
227         this.removeCallbacks = [];
228 
229 
230         /**
231          * @desc Inspect if item contains data
232          * @name Item.hasData
233          * @function
234          * @returns {bool}
235          */
236         this.hasData = function(){
237             return !!this.data;
238         }
239 
240 
241         /**
242          * @desc Get ETag response header
243          * @name Item.etag
244          * @function
245          * @return {string}
246          */
247         this.etag = function(){
248             if ( !this.jqXHR ) return '';
249             var etag = this.jqXHR.getResponseHeader('ETag');
250             return ( etag ) ? etag : '';
251         }
252 
253 
254         /**
255          * @desc Get Last-Modified response header
256          * @name Item.modified
257          * @function
258          * @return {string}
259          */
260         this.modified = function(){
261             if ( !this.jqXHR ) return '';
262             var modified = this.jqXHR.getResponseHeader('Last-Modified');
263             return ( modified ) ? modified : '';
264         }
265 
266 
267         /**
268          * @desc Bind function to write event. Callbacks consume four parameters:
269          * the response body, Etag header (if present), Last-Modofied header
270          * (if present) and the jqXHR object.
271          * @name Item.onWrite
272          * @function
273          * @param {func} func
274          * @returns {Item} This item, for chaining purposes
275          */
276         this.onWrite = function(func){
277             this.writeCallbacks.push(func);
278             return this;
279         }
280 
281 
282         /**
283          * @desc Bind function to remove event. Callbacks consume four parameters:
284          * the response body, Etag header (if present), Last-Modofied header
285          * (if present) and the jqXHR object.
286          * @name Item.onRemove
287          * @function
288          * @param {func} func
289          * @returns {Item} This item, for chaining purposes
290          */
291         this.onRemove = function(func){
292             this.removeCallbacks.push(func);
293             return this;
294         }
295 
296 
297         /**
298          * @desc Fire a write event. Does not send requests or affect cache content.
299          * @name Item.triggerWrite
300          * @function
301          * @returns {Item} This item, for chaining purposes
302          */
303         this.triggerWrite = function(){
304             var item = this;
305             $.each(this.writeCallbacks, function(index, func){
306                 func(item.data, item.etag(), item.modified(), item.jqXHR);
307             });
308             return this;
309         }
310 
311 
312         /**
313          * @desc Fire a remove event. Does not send requests or affect cache content.
314          * @name Item.triggerRemove
315          * @function
316          * @returns {Item} This item, for chaining purposes
317          */
318         this.triggerRemove = function(){
319             var item = this;
320             $.each(this.removeCallbacks, function(index, func){
321                 func(item.data, item.etag(), item.modified(), item.jqXHR);
322             });
323             return this;
324         }
325 
326 
327         /**
328          * @desc Write new content to item. Triggers write event.
329          * @name Item.write
330          * @function
331          * @param {mixed} data
332          * @param {jqXHR} jqXHR
333          * @returns {Item} This item, for chaining purposes
334          */
335         this.write = function(data, jqXHR){
336             this.data = data;
337             this.jqXHR = jqXHR;
338             this.triggerWrite();
339             return this;
340         }
341 
342 
343         /**
344          * @desc Remove item from cache
345          * @name Item.remove
346          * @function
347          * @returns {void}
348          */
349         this.remove = function(){
350             this.triggerRemove();
351             delete $.rcache.cache[this.url];
352         }
353 
354 
355         /* AJAX */
356 
357 
358         /**
359          * @desc If item is not in cache a http GET request is sent. If item is
360          * already in cache the previous jqXHR object is returned.
361          * @name Item.get
362          * @function
363          * @param {dict} options Additional jQuery.ajax options
364          * @returns {jqXHR}
365          */
366         this.get = function(options){
367             if ( this.hasData() && this.jqXHR ) {
368                 
369                 //Perform update in the background
370                 if ( $.rcache.settings['update-on-read'] ) {
371                     this.update(options);
372                 }
373                 
374                 //Return jqXHR of item already in cache
375                 return this.jqXHR;
376             } else {
377                 //Create new ajax request
378                 return this.forceGet(options);
379             }
380         }
381 
382 
383         /**
384          * @desc Perform a conditional http GET request. If resource have
385          * been updated changes are written to the cache.
386          * @name Item.update
387          * @function
388          * @param {dict} options Additional jQuery.ajax options
389          * @returns {jqXHR}
390          */
391         this.update = function(options){
392             var opts = {
393                 type: 'GET',
394                 url: this.url,
395                 headers: {
396                     'If-None-Match': this.etag(),
397                     'If-Modified-Since': this.modified(),
398                 }
399             };
400 
401             $.each(opts.headers, function(index, value){
402                 if ( !value ) delete opts.headers[index];
403             });
404 
405             $.extend(true, opts, options);
406 
407             var jqXHR = $.rcache.getJqXHR(opts);
408             var item = this;
409 
410             jqXHR.done(function(body, status, jqXHR){
411                 //Write to item on success
412                 //304 == Not Modified
413                 if ( jqXHR.status == 200 ) {
414                     item.write(body, jqXHR);
415                 }
416             });
417 
418             return jqXHR;
419         }
420 
421 
422         /**
423          * @desc Force a http GET request, even if item is in cache
424          * @name Item.forceGet
425          * @function
426          * @param {dict} options Additional jQuery.ajax options
427          * @returns {jqXHR}
428          */
429         this.forceGet = function(options){
430             if ( !options ) options = {};
431 
432             var opts = {
433                 type: 'GET',
434                 url: this.url,
435                 headers: {
436                     'If-None-Match': '',
437                     'If-Modified': '',
438                     'Cache-Control': 'no-cache',
439                     'Pragma': 'no-cache'
440                 }
441             };
442 
443             $.extend(true, opts, options);
444 
445             var jqXHR = $.rcache.getJqXHR(opts);
446             var item = this;
447 
448             jqXHR.done(function(body, status, jqXHR){
449                 //Write to item on success
450                 item.write(body, jqXHR);
451             });
452 
453             return jqXHR;
454         }
455 
456 
457         /**
458          * @desc Delete item using http DELETE and remove from cache
459          * @name Item.del
460          * @function
461          * @param {dict} options Additional jQuery.ajax options
462          * @returns {jqXHR}
463          */
464         this.del = function(options){
465             if ( !options ) options = {};
466 
467             var opts = {
468                 type: 'DELETE',
469                 url: this.url,
470                 headers: {
471                     'If-Match': this.etag(),
472                     'If-Unmodified-Since': this.modified(),
473                 }
474             };
475 
476             $.each(opts.headers, function(index, value){
477                 if ( !value ) delete opts.headers[index];
478             });
479 
480             $.extend(true, opts, options);
481      
482             var jqXHR = $.rcache.getJqXHR(opts);
483             var item = this;
484 
485             jqXHR.done(function(body, status, jqXHR){
486                 //Remove from cache on success
487                 item.remove();
488             });
489             return jqXHR;
490         }
491 
492 
493         /**
494          * @desc Send item using http PUT. If the server returns status code 200
495          * (OK) and a response body the returned resource is written to cache.
496          * @name Item.put
497          * @function
498          * @param {mixed} data
499          * @param {dict} options Additional jQuery.ajax options
500          * @returns {jqXHR}
501          */
502         this.put = function(data, options){
503             if ( !options ) options = {};
504 
505             var opts = {
506                 type: 'PUT',
507                 url: this.url,
508                 data: data,
509                 headers: {
510                     'If-Match': this.etag(),
511                     'If-Unmodified-Since': this.modified(),
512                 }
513             };
514 
515             $.each(opts.headers, function(index, value){
516                 if ( !value ) delete opts.headers[index];
517             });
518 
519             $.extend(true, opts, options);
520 
521             var jqXHR = $.rcache.getJqXHR(opts);
522             var item = this;
523 
524             jqXHR.done(function(body, status, jqXHR){
525                 //write to cache if these requirements are met
526                 if ( jqXHR.status == 200 && body ) {
527                     item.write(body, jqXHR);
528                 }
529             });
530 
531             return jqXHR;
532         }
533 
534 
535         /**
536          * @desc Send http POST request. If the server returns status code 200
537          * (OK) or 201 (created), a response body and a Content-Location header
538          * the resource is written to cache. Else if the server returns a
539          * Location header a GET request for that url is triggered.
540          * @name Item.post
541          * @function
542          * @param {mixed} data
543          * @param {dict} options Additional jQuery.ajax options
544          * @returns {jqXHR}
545          */
546         this.post = function(data, options){
547             if ( !options ) options = {};
548 
549             var opts = {
550                 type: 'POST',
551                 url: this.url,
552                 data: data
553             };
554 
555             $.extend(true, opts, options);
556      
557             var jqXHR = $.rcache.getJqXHR(opts);
558 
559             jqXHR.done(function(body, status, jqXHR){
560                 var contentLocation = jqXHR.getResponseHeader('Content-Location');
561                 var location = jqXHR.getResponseHeader('Location');
562 
563                 //write to cache if these requirements are met
564                 if (
565                     (jqXHR.status == 200 || jqXHR.status == 201)
566                     && body
567                     && contentLocation
568                 ) {
569                     $.rcache.item(contentLocation).write(body, jqXHR);
570                 
571                 //else get a fresh copy
572                 } else if ( location ) {
573                     $.rcache.item(location).forceGet();
574                 }
575             });
576             
577             return jqXHR;
578         }
579 
580     }
