 22     /**
 23      * @desc Active and observable ajax cache for jQuery REST applications. Uses
 24      * jQuery.ajax when interacting with the server.
 25      * <p>A note on keys: Cache keys are case sensitive urls. Take care to use
 26      * consistent urls, even query parameters and hashes count.</p>
 27      * <p>rcache only cache 200 and 201 responses. Redirects are not followed
 28      * due to the risk of creating redirect loops. 404 and 410 responses
 29      * automatically removes item from cache. 205 responses resets item to
 30      * current 'reset-resource-to' value (see setUp()).</p>
 31      * @author Hannes Forsgård
 32      * @class
 33      */
 34     $.rcache = new function(){
 35 
 36 
 37         /* Settings */
 38 
 39 
 40         /**
 41          * @desc rcache settings. See setUp() for altering settings.
 42          * @var {object} settings
 43          */
 44         this.settings = {
 45             'update-on-read': true,
 46             'reset-resource-to': {},
 47         };
 48 
 49 
 50         /**
 51          * @desc Set rcache settings. The following settings are available:
 52          * <dl>
 53          *     <dt>update-on-read</dt>
 54          *     <dd>Boolean. Send conditional get requests in the
 55          *     background on cache reads? Deafaults to true.</dd>
 56          *     <dt>reset-resource-to</dt>
 57          *     <dd>When the server responds with a 205 Reset content status code
 58          *     item data is reset to this value. Defaults to {}.</dd>
 59          * </dl>
 60          * @param {object} settings
 61          * @returns {$.rcache} Return rcache instance for chaining purposes
 62          */
 63         this.setUp = function(settings){
 64             $.extend(this.settings, settings);
 65             return this;
 66         }
 67 
 68         
 69         /* Basic functionality */        
 70         
 71 
 72         /**
 73          * @desc The internal cache store
 74          * @var {object} cache
 75          */
 76         this.cache = {};
 77 
 78 
 79         /**
 80          * @desc Get number of items in cache
 81          * @returns {int}
 82          */
 83         this.count = function(){
 84             var count = 0;
 85             $.each(this.cache, function() { 
 86                 count++;
 87             });
 88             return count;
 89         }
 90 
 91 
 92         /**
 93          * @desc Returns TRUE if cache includes item, FALSE otherwise
 94          * @param {string} key
 95          * @returns {bool}
 96          */
 97         this.has = function(key){
 98             return key in this.cache;
 99         }
100 
101 
102         /**
103          * @desc Get item for cache key. If item does not exist
104          * an empty item is created.
105          * @param {string} key
106          * @returns {Item}
107          */
108         this.item = function(key){
109             if ( !this.has(key) ) {
110                 this.cache[key] = new Item(key);
111             }
112             return this.cache[key];
113         }
114 
115 
116         /**
117          * @desc Clear cache
118          * @returns {$.rcache} Return rcache instance for chaining purposes
119          */
120        this.clear = function(){
121             this.cache = {};
122             return this;
123         }
124 
125 
126         /* AJAX */
127 
128 
129         /**
130          * @desc Update all items in cache where autoUpdate == true, using
131          * conditional GET requests
132          * @param {object} options Additional jQuery.ajax options
133          * @returns {$.rcache} Return rcache instance for chaining purposes
134          */
135         this.updateAll = function(options){
136             $.each(this.cache, function(key, item) { 
137                 if ( item.autoUpdate ) {
138                     item.update(options);
139                 }
140             });
141             return this;
142         }
143 
144 
145         /**
146          * @desc Settings used when creating ajax objects
147          * @var {object} ajaxOpts
148          */
149         this.ajaxOpts = {
150             // Default settings goes here ...
151             cache: false
152         }; 
153 
154 
155         /**
156          * @desc Set default ajax optins. By default rcache requests are created
157          * with the 'cache' option set to false.
158          * @param {object} options
159          * @returns {$.rcache} Return rcache instance for chaining purposes
160          */
161         this.ajaxSetup = function(options){
162             $.extend(this.ajaxOpts, options);
163             return this;
164         }
165 
166 
167         /**
168          * @desc Internal method to create and fire a jqXHR. On 404 Not Found or
169          * 410 Gone responses the cache item is deleted. On 205 Reset Content
170          * responses cache item is cleared to setting 'reset-resource-to'.
171          * @param {object} options
172          * @returns {jqXHR}
173          */
174         this.getJqXHR = function(options){
175             var settings = {};
176             $.extend(settings, this.ajaxOpts, options);
177             var jqXHR = $.ajax(settings);
178 
179             jqXHR.fail(function(jqXHR){
180                 // Remove from cache on 404 and 410 fails
181                 if ( jqXHR.status == 404 || jqXHR.status == 410 ) {
182                     $.rcache.item(settings.url).remove();
183                 }
184             });
185 
186             jqXHR.done(function(x, xx, jqXHR){
187                 // Reset cache content on 205 response
188                 if ( jqXHR.status == 205 ) {
189                     var empty = $.rcache.settings['reset-resource-to'];
190                     $.rcache.item(settings.url).write(empty, jqXHR);
191                 }
192             });
193 
194             return jqXHR;
195         }
196 
197     }

203     /**
204      * @desc Each item in cache is represented by an Item object
205      * @param {string} url Url of created item
206      * @class
207      * @name Item
208      */
209     function Item(url){
210 
211         /**
212          * @desc Item url
213          * @name Item.url
214          * @type string
215          */
216         this.url = url;
217 
218 
219         /**
220          * @desc Resource item data
221          * @name Item.data
222          * @type mixed
223          */
224         this.data = false;
225 
226  
227         /**
228          * @desc Request object that fetched resource
229          * @name Item.jqXHR
230          * @type jqXHR
231          */
232         this.jqXHR = false;
233 
234         
235         /**
236          * @desc If true this item will be updated on updateAll calls.
237          * @name Item.autoUpdate
238          * @type bool
239          */
240         this.autoUpdate = false;
241 
242 
243         /**
244          * @desc Callback functions for write events
245          * @name Item.writeCallbacks
246          * @type array
247          */
248         this.writeCallbacks = [];
249         
250 
251         /**
252          * @desc Callback functions for remove events
253          * @name Item.removeCallbacks
254          * @type array
255          */
256         this.removeCallbacks = [];
257     }

263     /**
264      * @desc Inspect if item contains data
265      * @name Item.hasData
266      * @function
267      * @returns {bool}
268      */
269     Item.prototype.hasData = function(){
270         return !!this.data;
271     }

274     /**
275      * @desc Get ETag response header
276      * @name Item.etag
277      * @function
278      * @return {string}
279      */
280     Item.prototype.etag = function(){
281         if ( !this.jqXHR ) return '';
282         var etag = this.jqXHR.getResponseHeader('ETag');
283         return ( etag ) ? etag : '';
284     }
285 
286 
287     /**
288      * @desc Get Last-Modified response header
289      * @name Item.modified
290      * @function
291      * @return {string}
292      */
293     Item.prototype.modified = function(){
294         if ( !this.jqXHR ) return '';
295         var modified = this.jqXHR.getResponseHeader('Last-Modified');
296         return ( modified ) ? modified : '';
297     }

300     /**
301      * @desc Bind function to write event. Callbacks consume four parameters:
302      * the response body, Etag header (if present), Last-Modofied header
303      * (if present) and the jqXHR object.
304      * @name Item.onWrite
305      * @function
306      * @param {func} func
307      * @returns {Item} This item, for chaining purposes
308      */
309     Item.prototype.onWrite = function(func){
310         this.writeCallbacks.push(func);
311         return this;
312     }
313 
314 
315     /**
316      * @desc Bind function to remove event. Callbacks consume four parameters:
317      * the response body, Etag header (if present), Last-Modofied header
318      * (if present) and the jqXHR object.
319      * @name Item.onRemove
320      * @function
321      * @param {func} func
322      * @returns {Item} This item, for chaining purposes
323      */
324     Item.prototype.onRemove = function(func){
325         this.removeCallbacks.push(func);
326         return this;
327     }

330     /**
331      * @desc Fire a write event. Does not send requests or affect cache content.
332      * @name Item.triggerWrite
333      * @function
334      * @returns {Item} This item, for chaining purposes
335      */
336     Item.prototype.triggerWrite = function(){
337         var item = this;
338         $.each(this.writeCallbacks, function(index, func){
339             func(item.data, item.etag(), item.modified(), item.jqXHR);
340         });
341         return this;
342     }

345     /**
346      * @desc Fire a remove event. Does not send requests or affect cache content.
347      * @name Item.triggerRemove
348      * @function
349      * @returns {Item} This item, for chaining purposes
350      */
351     Item.prototype.triggerRemove = function(){
352         var item = this;
353         $.each(this.removeCallbacks, function(index, func){
354             func(item.data, item.etag(), item.modified(), item.jqXHR);
355         });
356         return this;
357     }

360     /**
361      * @desc Write new content to item. Triggers write event. Sets
362      * autoUpdate to true, enabling this item to be updated on updateAll
363      * @name Item.write
364      * @function
365      * @param {mixed} data
366      * @param {jqXHR} jqXHR
367      * @returns {Item} This item, for chaining purposes
368      */
369     Item.prototype.write = function(data, jqXHR){
370         this.data = data;
371         this.jqXHR = jqXHR;
372         this.autoUpdate = true;
373         this.triggerWrite();
374         return this;
375     }

379      * @desc Remove item from cache
380      * @name Item.remove

382      * @returns {void}

384     Item.prototype.remove = function(){
385         this.triggerRemove();
386         delete $.rcache.cache[this.url];

393     /**
394      * @desc If item is not in cache a http GET request is sent. If item is
395      * already in cache the previous jqXHR object is returned.
396      * @name Item.get
397      * @function
398      * @param {object} options Additional jQuery.ajax options
399      * @returns {jqXHR}
400      */
401     Item.prototype.get = function(options){
402         if ( this.hasData() && this.jqXHR ) {
403             // Perform update in the background
404             if ( $.rcache.settings['update-on-read'] ) {
405                 this.update(options);
406             }
407             // Return jqXHR of item already in cache
408             return this.jqXHR;
409         } else {
410             // Create new ajax request
411             return this.forceGet(options);
412         }
413     }

416     /**
417      * @desc Perform a conditional http GET request. If resource have
418      * been updated changes are written to the cache.
419      * @name Item.update
420      * @function
421      * @param {object} options Additional jQuery.ajax options
422      * @returns {jqXHR}
423      */
424     Item.prototype.update = function(options){
425         var opts = {
426             type: 'GET',
427             url: this.url,
428             headers: {
429                 'If-None-Match': this.etag(),
430                 'If-Modified-Since': this.modified(),
431             }
432         };
433 
434         $.each(opts.headers, function(index, value){
435             if ( !value ) delete opts.headers[index];
436         });
437 
438         $.extend(true, opts, options);
439 
440         var jqXHR = $.rcache.getJqXHR(opts);
441         var item = this;
442 
443         jqXHR.done(function(body, status, jqXHR){
444             // Write to item on success
445             // 304 == Not Modified
446             if ( jqXHR.status == 200 ) {
447                 item.write(body, jqXHR);
448             }
449         });
450 
451         return jqXHR;
452     }

455     /**
456      * @desc Force a http GET request, even if item is in cache
457      * @name Item.forceGet
458      * @function
459      * @param {object} options Additional jQuery.ajax options
460      * @returns {jqXHR}
461      */
462     Item.prototype.forceGet = function(options){
463         if ( !options ) options = {};
464 
465         var opts = {
466             type: 'GET',
467             url: this.url,
468             headers: {
469                 'If-None-Match': '',
470                 'If-Modified': '',
471                 'Cache-Control': 'no-cache',
472                 'Pragma': 'no-cache'
473             }
474         };
475 
476         $.extend(true, opts, options);
477 
478         var jqXHR = $.rcache.getJqXHR(opts);
479         var item = this;
480 
481         jqXHR.done(function(body, status, jqXHR){
482             // Write to item on success
483             if ( jqXHR.status == 200 ) {
484                 item.write(body, jqXHR);
485             }
486         });
487 
488         return jqXHR;
489     }

493      * @desc Delete item using http DELETE and remove from cache
494      * @name Item.del
495      * @function
496      * @param {object} options Additional jQuery.ajax options
497      * @returns {jqXHR}
498      */
499     Item.prototype.del = function(options){
500         if ( !options ) options = {};
501 
502         var opts = {
503             type: 'DELETE',
504             url: this.url,
505             headers: {
506                 'If-Match': this.etag(),
507                 'If-Unmodified-Since': this.modified(),
508             }
509         };
510 
511         $.each(opts.headers, function(index, value){
512             if ( !value ) delete opts.headers[index];
513         });
514 
515         $.extend(true, opts, options);
516  
517         var jqXHR = $.rcache.getJqXHR(opts);
518         var item = this;
519 
520         jqXHR.done(function(body, status, jqXHR){
521             // Remove from cache on success
522             if ( jqXHR.status == 200 ) {
523                 item.remove();
524             }
525         });
526         return jqXHR;
527     }

572     /**
573      * @desc Send http POST request. If the server returns status code 200
574      * (OK) or 201 (created), a response body and a Content-Location header
575      * the resource is written to cache. Else if the server returns a
576      * Location header a GET request for that url is triggered.
577      * @name Item.post
578      * @function
579      * @param {mixed} data
580      * @param {object} options Additional jQuery.ajax options
581      * @returns {jqXHR}
582      */
583     Item.prototype.post = function(data, options){
584         if ( !options ) options = {};
585 
586         var opts = {
587             type: 'POST',
588             url: this.url,
589             data: data
590         };
591 
592         $.extend(true, opts, options);
593  
594         var jqXHR = $.rcache.getJqXHR(opts);
595 
596         jqXHR.done(function(body, status, jqXHR){
597             var contentLocation = jqXHR.getResponseHeader('Content-Location');
598             var location = jqXHR.getResponseHeader('Location');
599 
600             // Write to cache if these requirements are met
601             if (
602                 (jqXHR.status == 200 || jqXHR.status == 201)
603                 && body
604                 && contentLocation
605             ) {
606                 $.rcache.item(contentLocation).write(body, jqXHR);
607             
608             // Else get a fresh copy
609             } else if ( location ) {
610                 $.rcache.item(location).forceGet();
611             }
612         });
613         
614         return jqXHR;
615     }
